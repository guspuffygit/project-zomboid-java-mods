    /* Tool Roll Recipes    */
    /* ==================== */

    /* Assemble Tool Roll: Hammer + Screwdriver + Saw + empty ToolRoll container → CSR Tool Roll */
    craftRecipe CSR_MakeToolRoll
    {
        timedAction = PutInBag,
        category = Tools,
        time = 50,
        Tags = InHandCraft;CanBeDoneInDark,
        Tooltip = Tooltip_CSR_CraftToolRoll,

        inputs
        {
            item 1 [Base.BallPeenHammer;Base.BallPeenHammerForged;Base.Hammer;Base.HammerForged] flags[IsNotDull],
            item 1 [Base.Screwdriver;Base.Screwdriver_Old] flags[IsNotDull],
            item 1 [Base.Saw] flags[IsNotDull],
            item 1 [Base.ToolRoll_Leather;Base.ToolRoll_Fabric] mappers[rollType] flags[IsEmpty;Prop2],
        }
        outputs
        {
            item 1 mapper:rollType,
        }
        itemMapper rollType
        {
            Base.CSR_ToolRoll_Leather = Base.ToolRoll_Leather,
            Base.CSR_ToolRoll_Fabric = Base.ToolRoll_Fabric,
        }
    }

    /* Disassemble Tool Roll: CSR Tool Roll → returns empty ToolRoll container to ground */
    craftRecipe CSR_DisassembleToolRoll
    {
        timedAction = PutInBag,
        category = Tools,
        time = 40,
        Tags = InHandCraft;CanBeDoneInDark,
        Tooltip = Tooltip_CSR_DisassembleToolSet,

        inputs
        {
            item 1 [Base.CSR_ToolRoll_Leather;Base.CSR_ToolRoll_Fabric] mappers[rollType],
        }
        outputs
        {
            item 1 mapper:rollType,
            item 1 Base.Saw,
            item 1 Base.Hammer,
            item 1 Base.Screwdriver,
        }
        itemMapper rollType
        {
            Base.ToolRoll_Leather = Base.CSR_ToolRoll_Leather,
            Base.ToolRoll_Fabric = Base.CSR_ToolRoll_Fabric,
        }
    }
